**Step B: Create tray.js**

```javascript
import { getCurrentStep, holdPiece, getHeldPieceId, getPlacedThisStep } from './state.js';
import { showStepGhosts } from './ghost.js';
```

Module state:
```javascript
let _trayItemsEl = null;
```

Exports:

1. `initTray()` — Finds `.tray-items` element: `_trayItemsEl = document.querySelector('.tray-items')`.

2. `renderTray()` — Clears `_trayItemsEl.innerHTML = ''`. Gets `getCurrentStep()`. If null (build complete), return. Gets `getPlacedThisStep()` and `getHeldPieceId()`.

   For each piece in `step.pieces`:
   - Skip if `placedThisStep.has(piece.id)` (already placed — don't show in tray)
   - Create outer div: `const item = document.createElement('div'); item.className = 'tray-item'; item.dataset.pieceId = piece.id;`
   - Create inner swatch: `const swatch = document.createElement('div'); swatch.className = 'tray-swatch'; swatch.style.backgroundColor = piece.color;`
   - Set title for accessibility: `item.title = piece.type` (per UI-SPEC accessibility contract)
   - If `getHeldPieceId() === piece.id`, add `item.classList.add('selected')`
   - Click handler:
     ```javascript
     item.addEventListener('click', (e) => {
       e.stopPropagation();  // prevent canvas click-through
       holdPiece(piece.id);
       // Show all step ghosts (not just held piece — per RESEARCH.md OQ2 "show all simultaneously")
       const step = getCurrentStep();
       if (step) showStepGhosts(step);
       renderTray();  // re-render to update selected state
     });
     ```
   - Append swatch to item, item to `_trayItemsEl`

**Step C: Create hud.js**

```javascript
import { getCurrentStepNumber, getTotalSteps, getCurrentStep, isBuildComplete } from './state.js';
```

Module state:
```javascript
let _hudEl = null;
```

Exports:

1. `initHUD()` — Stores reference: `_hudEl = document.getElementById('hud')`.

2. `renderHUD()` — Clears `_hudEl.innerHTML = ''`. Checks state:

   **Loading state** (before set loaded — getCurrentStep() returns null and isBuildComplete() is false and getTotalSteps() is 0):
   ```html
   <div class="hud-step-counter">Loading set...</div>
   ```

   **Build complete state** (isBuildComplete() returns true):
   ```html
   <div class="hud-step-counter">Build complete!</div>
   ```

   **Normal state:**
   ```html
   <div class="hud-step-counter">Step {N} of {M}</div>
   <div class="hud-step-desc">{step.description}</div>
   <div class="hud-progress">
     <div class="hud-progress-fill" style="width: {percent}%"></div>
   </div>
   ```
   Where N = `getCurrentStepNumber()`, M = `getTotalSteps()`, step = `getCurrentStep()`, percent = `Math.round((N / M) * 100)`.

   Set `aria-live="polite"` on the `.hud-step-counter` element (per UI-SPEC accessibility contract).

**Step D: Refactor main.js** — Replace stress test with build loop.

New main.js content:
```javascript
import { initScene } from './scene.js';
import { loadSet, getCurrentStep } from './state.js';
import { showStepGhosts } from './ghost.js';
import { initInteraction } from './interaction.js';
import { initTray, renderTray } from './tray.js';
import { initHUD, renderHUD } from './hud.js';

const canvas = document.getElementById('canvas');
initScene(canvas);

// Initialize UI modules (DOM references)
initTray();
initHUD();

// Show loading state
renderHUD();

// Wire interaction callbacks
initInteraction({
  onStepAdvance: () => {
    renderTray();
    renderHUD();
  },
  onBuildComplete: () => {
    renderTray();
    renderHUD();
  },
});

// Fetch default set (mini-rocket — smallest set, per RESEARCH.md OQ1)
// Phase 3 will replace this hardcoded ID with set selection screen
async function startBuild() {
  try {
    const res = await fetch('/api/sets/mini-rocket');
    if (!res.ok) throw new Error(`Failed to load set: ${res.status}`);
    const setData = await res.json();
    loadSet(setData);
    renderTray();
    renderHUD();
    // Show ghosts for first step
    const firstStep = getCurrentStep();
    if (firstStep) showStepGhosts(firstStep);
  } catch (err) {
    console.error('Failed to start build:', err);
    document.getElementById('hud').innerHTML =
      '<div class="hud-step-counter">Error loading set</div>';
  }
}

startBuild();
```
